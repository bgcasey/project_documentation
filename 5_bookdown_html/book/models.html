2.4.4 Model plots

library(ggplot2)

# Create a scatterplot
ggplot(mtcars, aes(x=wt, y=mpg)) +
  geom_point() +
  geom_smooth(method='lm', se=FALSE) +
  ggtitle("Linear Regression of mpg vs wt")

ggsave("3_output/figures/regression.png")
